코드저장소

#include

int main(void) {
float money = 8, fee = 3;

printf("거래금액을 입력하세요: $ ");
scanf("%f", &money);

if (money < 2500) {
  fee = 30 + money * 0.017;
  if(fee < 39) fee = 39;
} else if ((money >= 2500) && (money < 6250)) {
  fee = 56 + money * 0.0066;
} else if ((money >= 6250) && (money < 20000)) {
  fee = 76 + money * 0.0034;
} else if ((money >= 50000) && (money < 500000)) {
  fee = 100 + money * 0.0022;
} else if ((money >= 20000) && (money < 50000)) {
  fee = 155 + money * 0.0011;
} else {
  fee = 255 + money * 0.0009;
}

printf("수수료는 $ %f 입니다.", fee);
return 0;
}
