int main(void) {
  int num[10] = {0};
  printf("Enter 10 numbers: ");
  for (int i = 0; i < 10; i++) {
    scanf("%d", &num[i]);
}

  int m, n = 0;
  max_min(num, 10, &m, &n);

  printf("Largest: %d\n", m);
  printf("Smallest: %d\n", n);

  return 0;
}

void max_min(int a[], int n, int *max, int *min) {

  *max = a[0];
  *min = a[0];

  int i = 0;

  while (i < n) {
    if (a[i] > *max) {
      *max = a[i];
    }
    if (a[i] < *min) {
      *min = a[i];
    }
    i++;
  }
}
