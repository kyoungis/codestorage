money = int(input("과세 표준 금액 입력(만원 단위): "))

if money <= 1200:
  tax = money * 0.06
elif money <= 4600:
 tax = 72 + (money - 1200)*0.15
elif money <= 8800:
 tax = 582 + (money - 4600)*0.24
elif money <= 15000:
 tax = 1590 + (money - 8800)*0.35
elif money <= 30000:
 tax = 3760 + (money - 15000)*0.38
elif money <= 50000:
  tax = 9460 + (money - 30000)*0.4
elif money <= 100000:
  tax = 17460 + (money - 50000)*0.42
else:
  tax = 38460 + (money - 100000)*0.45

tax = int(tax)

print("근로소득세액 : %d 만원" %tax)
